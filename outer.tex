\newlength{\CharCellHeight}
\newcommand{\Character}[5]{%
  \setbox0=\vbox{%
    \hsize=\linewidth
    \centering
    \resizebox{!}{0.74\linewidth}{\phantom{龠}}\par\vspace{2pt}%
    {\fontsize{11}{11}\selectfont\strut X}\par%
    \rule{0pt}{1ex}\par%
    \rule{0pt}{1ex}%
  }%
  \setlength{\CharCellHeight}{\ht0}%
  \addtolength{\CharCellHeight}{\dp0}%
  \begin{minipage}[t][\CharCellHeight]{\linewidth}%
    \centering
    \resizebox{!}{0.74\linewidth}{%
      \phantom{龠}%
      \llap{\hbox to \wd\CharProto{\hss#1\hss}}%
    }\par\vspace{2pt}%
    {\fontsize{11}{11}\selectfont\strut \RomanizationForm{#4}{#5}}\par% Romanisation
    \rule{0pt}{1ex}\begin{minipage}[t][0pt]{\linewidth}%
      \centering{\fontsize{10}{10}\selectfont\gray{#3}}% English meaning
    \end{minipage}\par%
    \rule{0pt}{1ex}\begin{minipage}[t][0pt]{\linewidth}%
      \centering{\fontsize{11}{11}\selectfont\gray{#2}}% Variant
    \end{minipage}%
  \end{minipage}%
  \par\vspace{4pt}%
}
