% Produces a block of exactly \CharCellHeight. Content pinned to top.
\newcommand{\Character}[5]{%
  \begin{minipage}[t][\CharCellHeight][t]{\linewidth}%
    \centering
    \resizebox{!}{0.74\linewidth}{%
      \cfont{\phantom{龠}}%
      \llap{\hbox to \wd\CharProto{\hss\cfont{#1}\hss}}%
    }\par\vspace{2pt}%
    {\fontsize{11}{11}\selectfont\strut \RomanizationForm{#4}{#5}}\par%
    \rule{0pt}{1ex}%
    \begin{minipage}[t][0pt]{\linewidth}%
      \centering\spaceskip=0.5em plus 0.5em minus 0.25em
      {\fontsize{10}{10}\selectfont\gray{\textls[-50]{#3}}}%
    \end{minipage}\par%
    \rule{0pt}{1ex}%
    \begin{minipage}[t][0pt]{\linewidth}%
      \centering{\fontsize{11}{11}\selectfont\gray{\cfont{#2}}}%
    \end{minipage}%
  \end{minipage}%
}

% ── Compute derived lengths now that \paperheight/\paperwidth are correct ──
% tboxouterheight = \paperheight - \vmargin - \vmargin - \eps
\setlength{\tboxouterheight}{\paperheight}
\addtolength{\tboxouterheight}{-\vmargin}
\addtolength{\tboxouterheight}{-\vmargin}
\addtolength{\tboxouterheight}{-\eps}

% CharCellHeight = (\tboxouterheight - \boxrule - \boxrule + \eps) / 15 - \eps
\setlength{\CharCellHeight}{\tboxouterheight}
\addtolength{\CharCellHeight}{-\boxrule}
\addtolength{\CharCellHeight}{-\boxrule}
\addtolength{\CharCellHeight}{\eps}
\divide\CharCellHeight by 15
\addtolength{\CharCellHeight}{-\eps}

% tboxwidth = (\paperwidth - \hmargin - \hmargin - \hgap) / 2 - \boxrule - \boxrule
\setlength{\tboxwidth}{\paperwidth}
\addtolength{\tboxwidth}{-\hmargin}
\addtolength{\tboxwidth}{-\hmargin}
\addtolength{\tboxwidth}{-\hgap}
\divide\tboxwidth by 2
\addtolength{\tboxwidth}{-\boxrule}
\addtolength{\tboxwidth}{-\boxrule}
